CREATE VIEW vw_Monitoramento AS 
SELECT
	p.ID_PACIENTE,
 	p.NOME,
 	SC.DESCRICAO,
 	tt.CLASSIFICACAO

FROM
	PACIENTE p
	JOIN ACOMPANHAMENTO AC ON AC.ID_PACIENTE=p.ID_PACIENTE
	JOIN STATUS_ACOMPANHAMENTO SC ON SC.ID_STATUS_ACOMPANHAMENTO = AC.ID_STATUSACOMPANHAMENTO
	JOIN TIPOTRATAMENTO TT ON TT.ID_TIPO_TRATAMENTO=AC.ID_TIPO_TRATAMENTO
